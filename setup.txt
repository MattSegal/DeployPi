===== Basic Linux Config ====

# Raspi config
sudo raspi-config

# Update then upgrade packages
	# download list of available packages, no installation
	sudo apt-get update

	# install new versions of packages
	sudo apt-get upgrade

===== Install Apps =====

# Install remote desktop
sudo apt-get install xrdp

# Install Samba [1]
sudo apt-get install samba

# Install kodi
sudo apt-get install kodi
